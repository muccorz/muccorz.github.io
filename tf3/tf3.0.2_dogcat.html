def build_model(hp):    

    for i in range(hp.Int('num_conv_layers',1,3)):
        model = tf.keras.models.Sequential()
        model.add(tf.keras.layers.Conv2D(hp.Choice(f'num_filters_layer{i}',values=[16,128],default=16),(3,3), 
                           activation = 'relu', input_shape = (150, 150, 3)))
        model.add(tf.keras.layers.MaxPooling2D(2,2))
        model.add(tf.keras.layers.Conv2D(hp.Choice(f'num_filters_layer{i}',values=[16,128],default=16), (3,3), activation = 'relu'))
        model.add(tf.keras.layers.MaxPooling2D(2,2))
    
    model.add(tf.keras.layers.Conv2D(64, (3,3), activation = 'relu'))
    model.add(tf.keras.layers.MaxPooling2D(2,2))
    model.add(tf.keras.layers.Flatten())
    model.add(tf.keras.layers.Dense(512, activation = 'relu'))
    model.add(tf.keras.layers.Dense(1, activation = 'sigmoid')) 
    model.compile(loss = 'binary_crossentropy',
             optimizer = tf.keras.optimizers.RMSprop(learning_rate=0.001),
             metrics=['acc'])
    return model
